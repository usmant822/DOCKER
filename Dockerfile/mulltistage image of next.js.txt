FROM node:tag-alpine as base
WORKDIR /app
COPY package.json ./
RUN npm install
COPY . .

FROM base as build 
ENV NODE_ENV=production
WORKDIR /build
COPY --from=base /app ./
RUN npm run build

FROM node:tag-alpine AS production
ENV NODE_ENV=production
WORDIR /app
COPY --from=build /build/package*.json ./
COPY --from=build /build/.next ./.next
COPY --from=build /build/public ./public
RUN npm install next

EXPOSE 3000
CMD["npm","run","start" ]
